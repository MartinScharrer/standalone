\def\maxerrorattest{}%

\makeatletter
\def\test#1{%
    \stepcounter{test}%
    \begingroup
    \savebox\boxa{{\includegraphics[#1]{testpic}}}
    \savebox\boxb{{\includegraphics[#1]{testpic.tex}}}
    %\savebox\boxc{{\includepicture [#1]{test}}}
    \begin{preview}
    \begin{tabular}{lll}
    \fbox{\usebox\boxa} & \fbox{\usebox\boxb} & \fbox{\usebox\boxc} \\
    \the\wd\boxa\ x \the\ht\boxa+\the\dp\boxa
    \message{^^JA: \the\wd\boxa\space x \the\ht\boxa+\the\dp\boxa}%
    &
    \the\wd\boxb\ x \the\ht\boxb+\the\dp\boxb
    \message{^^JB: \the\wd\boxb\space x \the\ht\boxb+\the\dp\boxb}%
    %&
    %\the\wd\boxc\ x \the\ht\boxc+\the\dp\boxc
    %\message{^^JC: \the\wd\boxc\space x \the\ht\boxc+\the\dp\boxc^^J}%
    \\
    \pgfmathsetlength\dW{abs(\wd\boxa - \wd\boxb)}%
    \pgfmathsetlength\dH{abs(\ht\boxa - \ht\boxb)}%
    \pgfmathsetlength\dD{abs(\dp\boxa - \dp\boxb)}%
    \ifdim\dW>\dWmax
        \global\dWmax=\dW
        \xdef\maxerrorattest{\the\value{test}}%
    \fi
    \ifdim\dH>\dHmax
        \global\dHmax=\dH
        \xdef\maxerrorattest{\the\value{test}}%
    \fi
    \ifdim\dD>\dDmax
        \global\dDmax=\dD
        \xdef\maxerrorattest{\the\value{test}}%
    \fi
    \edef\@tempa{%
    \ifdim\EPSILON<\dW
        dW = \the\dW\space
    \fi
    \ifdim\EPSILON<\dH
        dH = \the\dH\space
    \fi
    \ifdim\EPSILON<\dD
        dD = \the\dD\space
    \fi
    }%
    \edef\@tempa{%
    \ifx\@tempa\empty
        \ifcase 0%
            \ifdim\dD=\z@
            \ifdim\dW=\z@
            \ifdim\dH=\z@
                1%
            \fi
            \fi
            \fi
        \space
            \noexpand\OK
        \else
            \noexpand\PERFECT
        \fi
    \else
        \noexpand\FAIL{\@tempa}
    \fi}%
    \@tempa
    \def\OK{OK}
    \def\PERFECT{PERFECT}
    \def\FAIL##1{FAIL: ##1}
    \message{^^JTEST \the\value{test}: \@tempa^^J}%
    \\
    \end{tabular}
    \end{preview}
    \endgroup
}
\AtEndDocument{%
    \message{^^J%
        dWmax = \the\dWmax^^J%
        dHmax = \the\dHmax^^J%
        dDmax = \the\dDmax^^J%
        ^^J%
        }%
    \message{^^J%
        Results from \arabic{test} tests:^^J%
        PERFECT = \arabic{perfect}^^J%
        OK =      \arabic{ok}^^J%
        FAIL =    \arabic{fail}^^J%
        ^^J%
        Max error at test: \maxerrorattest^^J
        ^^J%
        }%
}
